            'interactionDescription': f'The unique functional group layout of this molecule displays zero known cross-reactivity rules against the chemical structure of {excipient}. This pairing is safe for formulation design.'
        })
        identified_evidence.append({
            'studyType': 'Preformulation Compatibility Screening',
            'compatibility': 'Compatible',
            'source': assigned_source,
            'details': assigned_details
        })

    return jsonify({
        'drugProfile': {
            'name': iupac_name, 'chemblId': chembl_id, 'molecularSpecies': 'Neutral',
            'maxPhase': '4.0', 'molecularWeight': molecular_weight, 'molecularFormula': molecular_formula, 'smiles': smiles
        },
        'excipientProfile': {
            'name': excipient, 'cas': excipient_details['cas'], 'formula': excipient_details['formula'],
            'synonyms': excipient_details['synonyms'], 'category': excipient_details['category']
        },
        'riskLevels': {'identifiedRisk': identified_risk, 'ruleBasedRisk': rule_based_risk},
        'identifiedEvidence': identified_evidence, 'ruleBasedEvidence': rule_based_evidence
    })

if __name__ == '__main__':
    app.run(debug=True, port=5000)

// Function to handle the calculation request
async function runPrediction() {
    const smiles = document.getElementById('smiles').value;
    const excipient = document.getElementById('excipient').value;
    const goBtn = document.getElementById('go-btn');

    if (!smiles.trim()) {
        alert("Please enter a valid Drug SMILES string.");
        return;
    }

    // Visual feedback for calculation state
    goBtn.innerText = "Computing...";
    goBtn.disabled = true;

    try {
        // Absolute full path URL to eliminate port ambiguity
        const res = await fetch('http://localhost:5000/api/predict', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ smiles: smiles, excipient: excipient })
        });
        
        if (!res.ok) {
            throw new Error(`Server returned status protocol: ${res.status}`);
        }

        const predictionData = await res.json();
        
        // Show the results section layout view
        document.getElementById('output-wrapper').classList.remove('hidden');

        // Populate Dynamic Profiles
        document.getElementById('d-name').innerText = predictionData.drugProfile.name;
        document.getElementById('d-chembl').innerText = predictionData.drugProfile.chemblId;
        document.getElementById('d-formula').innerText = predictionData.drugProfile.molecularFormula;
        document.getElementById('d-mw').innerText = predictionData.drugProfile.molecularWeight;
        document.getElementById('drug-img').src = `https://pubchem.ncbi.nlm.nih.gov/rest/pug/compound/smiles/${encodeURIComponent(predictionData.drugProfile.smiles)}/PNG`;

        document.getElementById('e-name').innerText = predictionData.excipientProfile.name;
        document.getElementById('e-cas').innerText = predictionData.excipientProfile.cas;
        document.getElementById('e-formula').innerText = predictionData.excipientProfile.formula;
        document.getElementById('e-cat').innerText = predictionData.excipientProfile.category;

        // Apply Risk Levels
        applyBadge('risk-ident', predictionData.riskLevels.identifiedRisk);
        applyBadge('risk-rule', predictionData.riskLevels.ruleBasedRisk);

        // Populate Identified Evidence Table Arrays
        const evTbody = document.getElementById('evidence-rows');
        evTbody.innerHTML = predictionData.identifiedEvidence.map(e => `
            <tr>
                <td><strong>${e.studyType}</strong></td>
                <td><span class="badge ${e.compatibility === 'Compatible' ? 'safe' : 'danger'}">${e.compatibility}</span></td>
                <td>${e.source}</td>
                <td>${e.details}</td>
            </tr>
        `).join('');

        // Populate Structural Incompatibility Rules
        const rTbody = document.getElementById('rules-rows');
        if (predictionData.ruleBasedEvidence.length === 0) {
            rTbody.innerHTML = `<tr><td colspan="4" style="color:#10b981; padding:15px; font-weight:600;">No structural risk rules triggered. The excipient combination is structurally stable.</td></tr>`;
        } else {
            rTbody.innerHTML = predictionData.ruleBasedEvidence.map(r => `
                <tr>
                    <td><strong>${r.drugRiskGroup}</strong> (<code>${r.drugGroupFormula}</code>)</td>
                    <td>${r.excipientRiskGroup}</td>
                    <td style="color:#ef4444; font-weight:bold">${r.reactionType}</td>
                    <td>${r.interactionDescription}</td>
                </tr>
            `).join('');
        }

    } catch (err) {
        console.error("Matrix Calculation Blocked:", err);
        alert(`Failed to complete calculation connection.\nVerify Python terminal is running at port 5000.\nError: ${err.message}`);
    } finally {
        goBtn.innerText = "GO!";
        goBtn.disabled = false;
    }
}

// Helper to style result assessment flags
function applyBadge(id, status) {
    const target = document.getElementById(id);
    if (!target) return;
    target.innerText = status;
    target.className = "badge " + (status.toLowerCase().includes('high') ? 'danger' : status.toLowerCase().includes('mod') ? 'warn' : 'safe');
}

// Tab Switching Mechanism
function switchTab(tabName) {
    document.querySelectorAll('.tab-pane').forEach(p => p.classList.add('hidden'));
    document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
    
    document.getElementById(`tab-${tabName}`).classList.remove('hidden');
    if (event && event.currentTarget) {
        event.currentTarget.classList.add('active');
    }
}
